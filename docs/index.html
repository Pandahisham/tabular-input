Tabular Input Fields

Simple Usage

jQuery('#tabular').tabularInput({
	'rows': 3,
	'columns': 5
});

Add / Delete Row Button

jQuery('#tabular').tabularInput({
	'rows': 3,
	'columns': 5
});
$("#tabular").tabularInput("addRow");
$("#tabular").tabularInput("deleteRow");

Auto Add Row on Tab

jQuery('#tabular').tabularInput({
	'rows': 3,
	'columns': 5,
	'newRowOnTab': true
});
